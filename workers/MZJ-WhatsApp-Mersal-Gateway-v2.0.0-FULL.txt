// MZJ Unified Platform - WhatsApp / Mersal Transport Gateway
// Version: 2.0.0
//
// This Worker is intentionally transport-only.
// It does NOT classify customers, create business rules, distribute leads,
// change CRM statuses, or run Inbox Agent decisions.
// Those responsibilities belong to MZJ Unified Platform.
//
// Routes:
//   GET  /                                      Health and route summary
//   GET  /health                                Health check
//   GET  /env-check                             Configuration flags without secret values
//   POST /webhooks/mersal/v1/messages           Mersal inbound webhook
//   POST /outbound/whatsapp/v1/text              Send free text / interactive buttons
//   POST /outbound/whatsapp/v1/template          Send an approved template
//   POST /outbound/whatsapp/v1/media             Send image/audio/video/document
//   GET|POST /templates/mersal/v1/sync           Fetch Mersal templates
//
// Compatibility aliases during migration:
//   POST /webhook/mersal
//   POST /send/mersal
//   GET|POST /templates/mersal
//
// Required Cloudflare secrets / variables:
//   MZJ_GATEWAY_SECRET                           Same value as Vercel
//   MERSAL_TOKEN                                 Mersal token (WA_TOKEN / MERSAL_API_TOKEN also accepted)
//   MZJ_PLATFORM_URL                             Example: https://mzj-platform.example.com
//
// Optional:
//   MZJ_PLATFORM_INBOUND_URL                     Full override for /api/integrations/whatsapp
//   MZJ_PLATFORM_MEDIA_PREPARE_URL               Full override for /api/integrations/media
//   MERSAL_WEBHOOK_SECRET                        Optional inbound webhook protection
//   MERSAL_API_ENDPOINT                          Default: https://w-mersal.com
//   MERSAL_SEND_URL                              Text send endpoint override
//   MERSAL_TEMPLATE_URL                          Template endpoint override
//   MERSAL_MEDIA_SEND_URL                        Media endpoint override
//   MERSAL_TEMPLATES_URL                         Templates endpoint override
//   MERSAL_API_TOKEN                             Token used by GetConversations/GetMessages
//   REQUEST_TIMEOUT_MS                           Default: 20000
//   MAX_MEDIA_BYTES                              Default: 100 MB
//
// Optional Queue binding:
//   MZJ_INBOUND_QUEUE
// When configured, webhook events are queued before processing. Without it,
// processing runs through ctx.waitUntil after the webhook is acknowledged.

const VERSION = "mzj-whatsapp-mersal-gateway-v2.0.0";
const SERVICE = "mzj-whatsapp-mersal-gateway";
const DEFAULT_MERSAL_BASE = "https://w-mersal.com";
const DEFAULT_TIMEOUT_MS = 20_000;
const DEFAULT_MAX_MEDIA_BYTES = 100 * 1024 * 1024;
const INBOUND_PATHS = new Set(["/webhooks/mersal/v1/messages", "/webhook/mersal"]);
const TEMPLATE_PATHS = new Set(["/templates/mersal/v1/sync", "/templates/mersal"]);

export default {
  async fetch(request, env, ctx) {
    const url = new URL(request.url);

    if (request.method === "OPTIONS") {
      return new Response(null, { status: 204, headers: corsHeaders() });
    }

    if (request.method === "GET" && (url.pathname === "/" || url.pathname === "/health")) {
      return json({
        ok: true,
        service: SERVICE,
        version: VERSION,
        mode: "transport_only",
        routes: {
          inbound: "/webhooks/mersal/v1/messages",
          text: "/outbound/whatsapp/v1/text",
          template: "/outbound/whatsapp/v1/template",
          media: "/outbound/whatsapp/v1/media",
          templatesSync: "/templates/mersal/v1/sync",
          envCheck: "/env-check",
        },
      });
    }

    if (request.method === "GET" && url.pathname === "/env-check") {
      return json({
        ok: true,
        service: SERVICE,
        version: VERSION,
        configured: {
          gatewaySecret: Boolean(getGatewaySecret(env)),
          mersalToken: Boolean(getMersalToken(env)),
          platformInbound: Boolean(platformInboundUrl(env)),
          platformMedia: Boolean(platformMediaPrepareUrl(env)),
          queue: Boolean(env?.MZJ_INBOUND_QUEUE && typeof env.MZJ_INBOUND_QUEUE.send === "function"),
          webhookSecret: Boolean(clean(env?.MERSAL_WEBHOOK_SECRET)),
          apiEndpoint: Boolean(mersalBaseUrl(env)),
        },
      });
    }

    if (INBOUND_PATHS.has(url.pathname)) {
      if (request.method !== "POST") return json({ ok: false, error: "Method not allowed" }, 405);
      const unauthorized = authorizeInboundWebhook(request, env);
      if (unauthorized) return unauthorized;
      const payload = await readJson(request);
      const envelope = {
        type: "mersal_webhook",
        receivedAt: new Date().toISOString(),
        payload,
      };
      if (env?.MZJ_INBOUND_QUEUE && typeof env.MZJ_INBOUND_QUEUE.send === "function") {
        await env.MZJ_INBOUND_QUEUE.send(envelope);
        return json({ ok: true, accepted: true, queued: true, version: VERSION }, 202);
      }
      const processing = processInboundEnvelope(envelope, env);
      if (ctx?.waitUntil) ctx.waitUntil(processing);
      else await processing;
      return json({ ok: true, accepted: true, queued: false, version: VERSION }, 202);
    }

    if (TEMPLATE_PATHS.has(url.pathname)) {
      if (request.method !== "POST" && request.method !== "GET") return json({ ok: false, error: "Method not allowed" }, 405);
      const unauthorized = authorizePlatformRequest(request, env);
      if (unauthorized) return unauthorized;
      return handleTemplates(env);
    }

    if (url.pathname === "/outbound/whatsapp/v1/text") {
      if (request.method !== "POST") return json({ ok: false, error: "Method not allowed" }, 405);
      const unauthorized = authorizePlatformRequest(request, env);
      if (unauthorized) return unauthorized;
      return sendText(await readJson(request), env);
    }

    if (url.pathname === "/outbound/whatsapp/v1/template") {
      if (request.method !== "POST") return json({ ok: false, error: "Method not allowed" }, 405);
      const unauthorized = authorizePlatformRequest(request, env);
      if (unauthorized) return unauthorized;
      return sendTemplate(await readJson(request), env);
    }

    if (url.pathname === "/outbound/whatsapp/v1/media") {
      if (request.method !== "POST") return json({ ok: false, error: "Method not allowed" }, 405);
      const unauthorized = authorizePlatformRequest(request, env);
      if (unauthorized) return unauthorized;
      return sendMedia(await readJson(request), env);
    }

    if (url.pathname === "/send/mersal") {
      if (request.method !== "POST") return json({ ok: false, error: "Method not allowed" }, 405);
      const unauthorized = authorizePlatformRequest(request, env);
      if (unauthorized) return unauthorized;
      const body = await readJson(request);
      if (clean(body?.mediaUrl || body?.media_url || body?.fileUrl || body?.file_url)) return sendMedia(body, env);
      if (clean(body?.template_name || body?.templateName || body?.template?.name)) return sendTemplate(body, env);
      return sendText(body, env);
    }

    return json({ ok: false, error: "Not found", version: VERSION }, 404);
  },

  async queue(batch, env) {
    for (const message of batch.messages) {
      try {
        await processInboundEnvelope(message.body || {}, env);
        message.ack();
      } catch (error) {
        console.error("MZJ inbound queue processing failed", safeError(error));
        message.retry({ delaySeconds: 60 });
      }
    }
  },
};

async function processInboundEnvelope(envelope, env) {
  const payload = envelope?.payload && typeof envelope.payload === "object" ? envelope.payload : envelope;
  const parsed = parseInboundWebhook(payload);
  if (!parsed.message) {
    console.log("MZJ Mersal webhook ignored: no inbound message");
    return { ok: true, ignored: true, reason: "no_message" };
  }

  let media = parsed.media;
  if (media?.hasAttachment) {
    media = await persistInboundMedia({ ...parsed, media }, env);
  }

  const event = {
    eventId: parsed.messageId,
    type: "message.received",
    direction: "in",
    senderType: "customer",
    source: "whatsapp",
    sourceCode: "whatsapp",
    platform: "whatsapp",
    channel: "whatsapp",
    provider: "mersal",
    participantId: parsed.phone,
    waId: parsed.phone,
    phone: parsed.phone,
    customerName: parsed.displayName,
    displayName: parsed.displayName,
    conversationId: parsed.phone,
    messageId: parsed.messageId,
    providerMessageId: parsed.messageId,
    messageType: parsed.messageType,
    text: parsed.text,
    caption: media?.caption || "",
    receivedAt: parsed.receivedAt,
    timestamp: parsed.timestamp,
    hasAttachment: Boolean(media?.hasAttachment),
    attachmentType: media?.type || "",
    mediaType: media?.type || "",
    fileName: media?.fileName || "",
    mimeType: media?.mimeType || "",
    fileSize: media?.fileSize || null,
    storageKey: media?.storageKey || "",
    mediaUrl: media?.storageKey ? "" : media?.url || "",
    mediaId: media?.mediaId || "",
    isSensitive: Boolean(media?.isSensitive),
    mediaStorageError: media?.storageError || "",
    rawEventReceivedAt: envelope?.receivedAt || new Date().toISOString(),
  };

  return postPlatformEvent(event, env);
}

function parseInboundWebhook(payload) {
  const value = payload?.entry?.[0]?.changes?.[0]?.value || payload?.value || payload;
  const message = value?.messages?.[0] || payload?.messages?.[0] || payload?.message || (payload?.from && payload?.type ? payload : null);
  if (!message) return { message: null };
  const contacts = value?.contacts || payload?.contacts || [];
  const contact = contacts[0] || payload?.contact || {};
  const phone = normalizeSaudiPhone(message?.from || contact?.wa_id || payload?.waId || payload?.phone);
  if (!phone) throw new Error("Inbound WhatsApp phone is missing or invalid");
  const displayName = clean(contact?.profile?.name || payload?.displayName || payload?.customerName || "عميل");
  const messageId = clean(message?.id || payload?.messageId || payload?.message_id) || `mersal_${Date.now()}_${randomId()}`;
  const messageType = clean(message?.type || payload?.messageType || "text").toLowerCase();
  const media = extractInboundMedia(message, messageType);
  const text = clean(
    message?.text?.body ||
    message?.button?.text ||
    message?.interactive?.button_reply?.title ||
    message?.interactive?.list_reply?.title ||
    media?.caption ||
    payload?.text ||
    payload?.message ||
    "",
  );
  const timestamp = clean(message?.timestamp || payload?.timestamp);
  const receivedAt = timestamp && /^\d+$/.test(timestamp)
    ? new Date(Number(timestamp) * 1000).toISOString()
    : new Date().toISOString();
  return { payload, value, message, contact, phone, displayName, messageId, messageType, text, timestamp, receivedAt, media };
}

function extractInboundMedia(message, messageType) {
  const type = normalizeMediaType(messageType);
  const media = message?.[messageType] || message?.[type];
  if (!media || !["image", "audio", "video", "document", "sticker"].includes(type)) return null;
  const mediaId = clean(media?.id);
  const url = clean(media?.url || media?.link || media?.href);
  const mimeType = clean(media?.mime_type || media?.mimeType) || guessMimeType(url, type);
  const fileName = clean(media?.filename || media?.fileName || media?.name) || buildFileName(type, mediaId, mimeType);
  return {
    hasAttachment: true,
    type,
    mediaId,
    url,
    mimeType,
    fileName,
    caption: clean(media?.caption),
    sha256: clean(media?.sha256),
    isSensitive: isSensitiveFile(fileName),
  };
}

async function persistInboundMedia(parsed, env) {
  const media = { ...parsed.media };
  try {
    let resolvedUrl = clean(media.url);
    if (!resolvedUrl || isProtectedWhatsappUrl(resolvedUrl)) {
      const resolved = await resolveMersalMedia(env, {
        phone: parsed.phone,
        messageId: parsed.messageId,
        mediaId: media.mediaId,
        messageType: media.type,
      });
      if (resolved?.url) {
        resolvedUrl = resolved.url;
        media.url = resolved.url;
        media.mimeType = media.mimeType || resolved.mimeType;
        media.fileName = media.fileName || resolved.fileName;
      }
    }
    if (!resolvedUrl) throw new Error("Mersal media URL was not resolved");

    const mediaResponse = await fetchWithTimeout(resolvedUrl, { method: "GET", headers: { accept: "*/*" } }, env, 60_000);
    if (!mediaResponse.ok) throw new Error(`Media download failed: HTTP ${mediaResponse.status}`);
    const bytes = await mediaResponse.arrayBuffer();
    const maxBytes = maximumMediaBytes(env);
    if (!bytes.byteLength) throw new Error("Downloaded media is empty");
    if (bytes.byteLength > maxBytes) throw new Error(`Media exceeds configured limit (${maxBytes} bytes)`);
    const mimeType = clean(mediaResponse.headers.get("content-type")) || media.mimeType || guessMimeType(resolvedUrl, media.type);
    const fileName = media.fileName || fileNameFromUrl(resolvedUrl) || buildFileName(media.type, media.mediaId, mimeType);

    const prepared = await preparePlatformMedia({
      source: "whatsapp",
      eventKey: parsed.messageId,
      fileName,
      mimeType,
      fileSize: bytes.byteLength,
      isSensitive: media.isSensitive || isSensitiveFile(fileName),
    }, env);

    const upload = await fetchWithTimeout(prepared.uploadUrl, {
      method: "PUT",
      headers: { "content-type": mimeType || "application/octet-stream" },
      body: bytes,
    }, env, 120_000);
    if (!upload.ok) throw new Error(`R2 upload failed: HTTP ${upload.status}`);

    return {
      ...media,
      url: "",
      storageKey: prepared.storageKey,
      assetId: prepared.assetId || "",
      fileName,
      mimeType,
      fileSize: bytes.byteLength,
      isSensitive: media.isSensitive || isSensitiveFile(fileName),
      storageError: "",
    };
  } catch (error) {
    console.error("Inbound media persistence failed", safeError(error));
    throw new Error(`Inbound media persistence failed: ${errorMessage(error, "تعذر حفظ الوسائط الواردة")}`);
  }
}

async function preparePlatformMedia(input, env) {
  const url = platformMediaPrepareUrl(env);
  if (!url) throw new Error("MZJ platform media URL is not configured");
  const response = await platformFetch(url, {
    method: "POST",
    body: JSON.stringify({ action: "prepare_upload", ...input }),
  }, env);
  const parsed = await parseResponse(response);
  if (!response.ok || parsed.data?.ok === false || !clean(parsed.data?.uploadUrl) || !clean(parsed.data?.storageKey)) {
    throw new Error(providerError(parsed.data, parsed.text, `Platform media prepare failed: HTTP ${response.status}`));
  }
  return parsed.data;
}

async function postPlatformEvent(event, env) {
  const url = platformInboundUrl(env);
  if (!url) throw new Error("MZJ platform inbound URL is not configured");
  let lastError = "Platform inbound request failed";
  for (let attempt = 1; attempt <= 3; attempt += 1) {
    try {
      const response = await platformFetch(url, { method: "POST", body: JSON.stringify(event) }, env);
      const parsed = await parseResponse(response);
      if (response.ok && parsed.data?.ok !== false) return { ok: true, status: response.status, data: parsed.data };
      lastError = providerError(parsed.data, parsed.text, `HTTP ${response.status}`);
    } catch (error) {
      lastError = errorMessage(error, lastError);
    }
    if (attempt < 3) await sleep(attempt * 800);
  }
  throw new Error(lastError);
}

async function sendText(body, env) {
  const token = requireMersalToken(env);
  if (token.error) return token.error;
  const phone = normalizeSaudiPhone(body?.phone || body?.waId || body?.mobile || body?.to);
  const message = clean(body?.message || body?.text);
  if (!phone) return json({ ok: false, error: "رقم واتساب غير صالح" }, 400);
  if (!message) return json({ ok: false, error: "نص الرسالة مطلوب" }, 400);
  const endpoint = clean(env?.MERSAL_SEND_URL) || `${mersalBaseUrl(env)}/api/wpbox/sendmessage`;
  const payload = { token: token.value, phone, message };
  if (Array.isArray(body?.buttons) && body.buttons.length) payload.buttons = body.buttons;
  if (clean(body?.header)) payload.header = clean(body.header);
  if (clean(body?.footer)) payload.footer = clean(body.footer);
  return callMersal(endpoint, payload, { messageType: "text", phone }, env);
}

async function sendTemplate(body, env) {
  const token = requireMersalToken(env);
  if (token.error) return token.error;
  const phone = normalizeSaudiPhone(body?.phone || body?.waId || body?.mobile || body?.to);
  const templateName = clean(body?.template_name || body?.templateName || body?.template?.name);
  if (!phone) return json({ ok: false, error: "رقم واتساب غير صالح" }, 400);
  if (!templateName) return json({ ok: false, error: "اسم القالب مطلوب" }, 400);
  const language = clean(body?.template_language || body?.templateLanguage || body?.templateLang || body?.template?.lang || "ar") || "ar";
  const components = resolveTemplateComponents(body);
  const endpoint = clean(env?.MERSAL_TEMPLATE_URL) || `${mersalBaseUrl(env)}/api/wpbox/sendtemplatemessage`;
  const payload = { token: token.value, phone, template_name: templateName, template_language: language };
  if (components.length) payload.components = components;
  return callMersal(endpoint, payload, { messageType: "template", phone, templateName, templateLanguage: language }, env);
}

async function sendMedia(body, env) {
  const token = requireMersalToken(env);
  if (token.error) return token.error;
  const phone = normalizeSaudiPhone(body?.phone || body?.waId || body?.mobile || body?.to);
  const mediaUrl = clean(body?.mediaUrl || body?.media_url || body?.fileUrl || body?.file_url || body?.url);
  const mediaType = normalizeMediaType(body?.mediaType || body?.media_type || body?.attachmentType || body?.type || mimeToMediaType(body?.mimeType || body?.mime_type));
  if (!phone) return json({ ok: false, error: "رقم واتساب غير صالح" }, 400);
  if (!mediaUrl) return json({ ok: false, error: "رابط الوسائط مطلوب" }, 400);
  if (!["image", "audio", "video", "document"].includes(mediaType)) return json({ ok: false, error: "نوع الوسائط يجب أن يكون image أو audio أو video أو document" }, 400);
  const endpoint = clean(env?.MERSAL_MEDIA_SEND_URL) || `${mersalBaseUrl(env)}/api/wpbox/sendmedia`;
  const payload = {
    token: token.value,
    phone,
    type: mediaType,
    media_url: mediaUrl,
  };
  const caption = clean(body?.caption || body?.message || body?.text);
  const fileName = clean(body?.fileName || body?.file_name);
  if (caption) payload.caption = caption;
  if (mediaType === "document" && fileName) payload.filename = fileName;
  return callMersal(endpoint, payload, { messageType: "media", mediaType, phone, fileName }, env);
}

async function handleTemplates(env) {
  const token = requireMersalToken(env);
  if (token.error) return token.error;
  const endpoint = clean(env?.MERSAL_TEMPLATES_URL) || `${mersalBaseUrl(env)}/api/wpbox/getTemplates?token=${encodeURIComponent(token.value)}`;
  try {
    const response = await fetchWithTimeout(endpoint, { method: "GET", headers: { accept: "application/json" } }, env);
    const parsed = await parseResponse(response);
    const templates = extractTemplates(parsed.data);
    const ok = response.ok && !providerSaysFailure(parsed.data);
    return json({
      ok,
      service: SERVICE,
      version: VERSION,
      provider: "mersal",
      count: templates.length,
      templates,
      error: ok ? undefined : providerError(parsed.data, parsed.text, `HTTP ${response.status}`),
    }, ok ? 200 : 502);
  } catch (error) {
    return json({ ok: false, service: SERVICE, version: VERSION, provider: "mersal", error: errorMessage(error, "تعذر جلب القوالب") }, 502);
  }
}

async function callMersal(endpoint, payload, meta, env) {
  try {
    const response = await fetchWithTimeout(endpoint, {
      method: "POST",
      headers: { "content-type": "application/json; charset=utf-8", accept: "application/json" },
      body: JSON.stringify(payload),
    }, env);
    const parsed = await parseResponse(response);
    const ok = response.ok && !providerSaysFailure(parsed.data);
    const providerMessageId = clean(parsed.data?.message_wamid || parsed.data?.wamid || parsed.data?.message_id || parsed.data?.id);
    return json({
      ok,
      service: SERVICE,
      version: VERSION,
      provider: "mersal",
      ...meta,
      httpStatus: response.status,
      status: providerStatus(parsed.data, ok),
      providerMessageId: providerMessageId || undefined,
      raw: parsed.data,
      error: ok ? undefined : providerError(parsed.data, parsed.text, `HTTP ${response.status}`),
    }, ok ? 200 : 502);
  } catch (error) {
    return json({ ok: false, service: SERVICE, version: VERSION, provider: "mersal", ...meta, error: errorMessage(error, "تعذر الاتصال بمرسال") }, 502);
  }
}

async function resolveMersalMedia(env, input) {
  const token = clean(env?.MERSAL_API_TOKEN || getMersalToken(env));
  if (!token) return { ok: false, error: "MERSAL_API_TOKEN is missing" };
  const contactId = await mersalFindContactId(env, token, input.phone);
  if (!contactId) return { ok: false, error: "Mersal contact not found" };
  const delays = [0, 900, 1600, 2500, 3500];
  for (const delay of delays) {
    if (delay) await sleep(delay);
    const payload = await mersalPost(env, `${mersalBaseUrl(env)}/api/wpbox/getMessages`, { token, contact_id: contactId });
    const rows = normalizeRows(payload);
    const found = findMersalMedia(rows, input);
    if (found?.url) return { ok: true, contactId, ...found };
  }
  return { ok: false, error: "Media URL not found in Mersal messages", contactId };
}

async function mersalFindContactId(env, token, phone) {
  const payload = await mersalPost(env, `${mersalBaseUrl(env)}/api/wpbox/getConversations/none?mobile_api=true`, { token });
  const rows = Array.isArray(payload?.data) ? payload.data : Array.isArray(payload) ? payload : [];
  const normalized = normalizeSaudiPhone(phone);
  const found = rows.find((row) => {
    const candidate = normalizeSaudiPhone(row?.phone || row?.name || "");
    return candidate && (candidate === normalized || candidate.endsWith(normalized.slice(-9)));
  });
  return clean(found?.id || found?.contact_id);
}

async function mersalPost(env, url, body) {
  try {
    const response = await fetchWithTimeout(url, {
      method: "POST",
      headers: { "content-type": "application/json", accept: "application/json" },
      body: JSON.stringify(body),
    }, env);
    const parsed = await parseResponse(response);
    if (response.ok && !isInvalidToken(parsed.data)) return parsed.data;
  } catch {}
  const form = new URLSearchParams();
  Object.entries(body || {}).forEach(([key, value]) => form.set(key, String(value ?? "")));
  const response = await fetchWithTimeout(url, {
    method: "POST",
    headers: { "content-type": "application/x-www-form-urlencoded", accept: "application/json" },
    body: form.toString(),
  }, env);
  return (await parseResponse(response)).data;
}

function normalizeRows(payload) {
  const rows = Array.isArray(payload) ? payload
    : Array.isArray(payload?.data) ? payload.data
    : Array.isArray(payload?.messages) ? payload.messages
    : Array.isArray(payload?.data?.messages) ? payload.data.messages
    : Array.isArray(payload?.data?.data) ? payload.data.data
    : [];
  return rows.slice().sort((a, b) => timestampValue(b?.created_at || b?.createdAt || b?.timestamp) - timestampValue(a?.created_at || a?.createdAt || a?.timestamp));
}

function findMersalMedia(rows, input) {
  const messageId = clean(input.messageId);
  const mediaId = clean(input.mediaId);
  const type = normalizeMediaType(input.messageType);
  if (messageId) {
    const row = rows.find((item) => clean(item?.fb_message_id || item?.message_wamid || item?.wamid || item?.message_id) === messageId);
    const result = mediaFromRow(row);
    if (result?.url) return result;
  }
  if (mediaId) {
    const result = rows.map(mediaFromRow).filter(Boolean).find((item) => clean(item.url).includes(mediaId) || clean(item.fileName).includes(mediaId));
    if (result?.url) return result;
  }
  const inbound = rows
    .filter((row) => Number(row?.is_message_by_contact || row?.isMessageByContact || 0) === 1)
    .map(mediaFromRow)
    .filter(Boolean);
  return inbound.find((item) => normalizeMediaType(item.type) === type) || inbound[0] || null;
}

function mediaFromRow(row) {
  if (!row) return null;
  const candidates = [
    ["image", row?.header_image || row?.image || row?.image_url || row?.media_image],
    ["audio", row?.header_audio || row?.audio || row?.audio_url || row?.voice || row?.voice_url || row?.media_audio],
    ["video", row?.header_video || row?.video || row?.video_url || row?.media_video],
    ["document", row?.header_document || row?.document || row?.document_url || row?.file || row?.file_url || row?.media_document],
  ];
  const candidate = candidates.find(([, value]) => clean(value));
  if (!candidate) return null;
  const type = candidate[0];
  const url = clean(candidate[1]);
  return { type, url, mimeType: guessMimeType(url, type), fileName: fileNameFromUrl(url) };
}

function authorizePlatformRequest(request, env) {
  const expected = getGatewaySecret(env);
  if (!expected) return json({ ok: false, error: "MZJ_GATEWAY_SECRET is not configured", version: VERSION }, 503);
  const provided = clean(request.headers.get("x-mzj-gateway-secret"));
  if (!safeEquals(provided, expected)) return json({ ok: false, error: "Unauthorized gateway request", version: VERSION }, 401);
  return null;
}

function authorizeInboundWebhook(request, env) {
  const expected = clean(env?.MERSAL_WEBHOOK_SECRET);
  if (!expected) return null;
  const url = new URL(request.url);
  const provided = clean(request.headers.get("x-mersal-webhook-secret") || url.searchParams.get("secret"));
  if (!safeEquals(provided, expected)) return json({ ok: false, error: "Unauthorized Mersal webhook", version: VERSION }, 401);
  return null;
}

function platformInboundUrl(env) {
  const override = clean(env?.MZJ_PLATFORM_INBOUND_URL);
  if (override) return override;
  const base = clean(env?.MZJ_PLATFORM_URL).replace(/\/+$/, "");
  return base ? `${base}/api/integrations/whatsapp` : "";
}

function platformMediaPrepareUrl(env) {
  const override = clean(env?.MZJ_PLATFORM_MEDIA_PREPARE_URL);
  if (override) return override;
  const base = clean(env?.MZJ_PLATFORM_URL).replace(/\/+$/, "");
  return base ? `${base}/api/integrations/media` : "";
}

function platformFetch(url, options, env) {
  return fetchWithTimeout(url, {
    ...options,
    headers: {
      "content-type": "application/json; charset=utf-8",
      accept: "application/json",
      "x-mzj-gateway-secret": getGatewaySecret(env),
      ...(options?.headers || {}),
    },
  }, env, 30_000);
}

function requireMersalToken(env) {
  const value = getMersalToken(env);
  return value ? { value } : { error: json({ ok: false, error: "MERSAL_TOKEN is not configured", version: VERSION }, 503) };
}

function getGatewaySecret(env) {
  return clean(env?.MZJ_GATEWAY_SECRET || env?.GATEWAY_SECRET);
}

function getMersalToken(env) {
  return clean(env?.MERSAL_TOKEN || env?.WA_TOKEN || env?.MERSAL_API_TOKEN);
}

function mersalBaseUrl(env) {
  return clean(env?.MERSAL_API_ENDPOINT || DEFAULT_MERSAL_BASE).replace(/\/+$/, "") || DEFAULT_MERSAL_BASE;
}

function maximumMediaBytes(env) {
  const value = Number(env?.MAX_MEDIA_BYTES || DEFAULT_MAX_MEDIA_BYTES);
  return Number.isFinite(value) && value > 0 ? Math.min(value, 100 * 1024 * 1024) : DEFAULT_MAX_MEDIA_BYTES;
}

function requestTimeout(env) {
  const value = Number(env?.REQUEST_TIMEOUT_MS || DEFAULT_TIMEOUT_MS);
  return Number.isFinite(value) ? Math.max(3_000, Math.min(60_000, value)) : DEFAULT_TIMEOUT_MS;
}

async function fetchWithTimeout(url, options, env, explicitTimeout) {
  const controller = new AbortController();
  const timeout = setTimeout(() => controller.abort("request_timeout"), explicitTimeout || requestTimeout(env));
  try {
    return await fetch(url, { ...options, signal: controller.signal });
  } finally {
    clearTimeout(timeout);
  }
}

async function parseResponse(response) {
  const text = await response.text();
  if (!text) return { text: "", data: {} };
  try { return { text, data: JSON.parse(text) }; }
  catch { return { text, data: { raw: text } }; }
}

function resolveTemplateComponents(body) {
  const direct = body?.components || body?.template?.components;
  if (Array.isArray(direct) && direct.length) return direct;
  const params = body?.params || body?.parameters || body?.template?.params || body?.template?.parameters;
  if (!Array.isArray(params) || !params.length) return [];
  return [{ type: "body", parameters: params.map((value) => value && typeof value === "object" && clean(value.type) ? value : { type: "text", text: clean(value) }) }];
}

function extractTemplates(payload) {
  if (Array.isArray(payload)) return payload;
  if (!payload || typeof payload !== "object") return [];
  return [payload.templates, payload.data, payload.result, payload.items, payload.response?.templates, payload.response?.data].find(Array.isArray) || [];
}

function providerSaysFailure(payload) {
  if (!payload || typeof payload !== "object") return false;
  if (payload.ok === false || payload.success === false) return true;
  return ["error", "failed", "failure", "unauthorized", "invalid"].includes(clean(payload.status).toLowerCase());
}

function providerStatus(payload, ok) {
  return clean(payload?.status || payload?.message_status || payload?.state) || (ok ? "success" : "error");
}

function providerError(payload, text, fallback) {
  return clean(payload?.error || payload?.message || payload?.detail || payload?.reason || text) || fallback;
}

function isInvalidToken(payload) {
  return clean(payload?.status).toLowerCase() === "error" && clean(payload?.message || payload?.errMsg).toLowerCase().includes("invalid token");
}

function normalizeSaudiPhone(value) {
  let phone = String(value || "").replace(/\D/g, "");
  if (phone.startsWith("00966")) phone = phone.slice(2);
  if (/^05\d{8}$/.test(phone)) phone = `966${phone.slice(1)}`;
  if (/^5\d{8}$/.test(phone)) phone = `966${phone}`;
  return /^9665\d{8}$/.test(phone) ? phone : "";
}

function normalizeMediaType(value) {
  const type = clean(value).toLowerCase();
  if (["photo", "picture"].includes(type)) return "image";
  if (["voice", "ptt"].includes(type)) return "audio";
  if (["file", "pdf"].includes(type)) return "document";
  return type;
}

function mimeToMediaType(value) {
  const mime = clean(value).toLowerCase();
  if (mime.startsWith("image/")) return "image";
  if (mime.startsWith("audio/")) return "audio";
  if (mime.startsWith("video/")) return "video";
  return "document";
}

function guessMimeType(url, type) {
  const lower = clean(url).toLowerCase().split("?")[0];
  const map = [
    [".jpg", "image/jpeg"], [".jpeg", "image/jpeg"], [".png", "image/png"], [".webp", "image/webp"], [".gif", "image/gif"],
    [".mp3", "audio/mpeg"], [".ogg", "audio/ogg"], [".opus", "audio/ogg"], [".wav", "audio/wav"], [".aac", "audio/aac"],
    [".mp4", "video/mp4"], [".webm", "video/webm"], [".pdf", "application/pdf"],
  ];
  const found = map.find(([extension]) => lower.endsWith(extension));
  if (found) return found[1];
  if (type === "image" || type === "sticker") return type === "sticker" ? "image/webp" : "image/jpeg";
  if (type === "audio") return "audio/mpeg";
  if (type === "video") return "video/mp4";
  return "application/octet-stream";
}

function extensionFromMime(mime, type) {
  const value = clean(mime).toLowerCase();
  const map = [
    ["jpeg", ".jpg"], ["jpg", ".jpg"], ["png", ".png"], ["webp", ".webp"], ["gif", ".gif"], ["pdf", ".pdf"],
    ["mp4", ".mp4"], ["webm", ".webm"], ["mpeg", type === "audio" ? ".mp3" : ".mpeg"], ["ogg", ".ogg"], ["wav", ".wav"], ["aac", ".aac"],
  ];
  const found = map.find(([needle]) => value.includes(needle));
  if (found) return found[1];
  return type === "image" ? ".jpg" : type === "audio" ? ".mp3" : type === "video" ? ".mp4" : type === "sticker" ? ".webp" : ".bin";
}

function buildFileName(type, id, mime) {
  const safeType = clean(type).replace(/[^a-z0-9_-]/gi, "") || "media";
  const safeId = clean(id).replace(/[^a-z0-9_.-]/gi, "").slice(0, 48) || `${Date.now()}_${randomId()}`;
  return `${safeType}-${safeId}${extensionFromMime(mime, type)}`;
}

function fileNameFromUrl(value) {
  try { return decodeURIComponent(new URL(clean(value)).pathname.split("/").pop() || ""); }
  catch { return ""; }
}

function timestampValue(value) {
  if (!value) return 0;
  if (typeof value === "number") return value < 1e12 ? value * 1000 : value;
  const parsed = Date.parse(String(value));
  return Number.isFinite(parsed) ? parsed : 0;
}

function safeEquals(left, right) {
  const a = new TextEncoder().encode(String(left || ""));
  const b = new TextEncoder().encode(String(right || ""));
  if (!a.length || a.length !== b.length) return false;
  let mismatch = 0;
  for (let index = 0; index < a.length; index += 1) mismatch |= a[index] ^ b[index];
  return mismatch === 0;
}

async function readJson(request) {
  const text = await request.text();
  if (!text) return {};
  try { return JSON.parse(text); }
  catch { return {}; }
}

function clean(value) {
  return String(value ?? "").trim();
}

function randomId() {
  return Math.random().toString(16).slice(2, 10);
}

function errorMessage(error, fallback) {
  if (error instanceof Error) {
    if (error.name === "AbortError") return "انتهت مهلة الاتصال";
    return error.message || fallback;
  }
  return clean(error) || fallback;
}

function safeError(error) {
  return { message: errorMessage(error, "Unknown error"), name: error?.name || "Error" };
}

function sleep(ms) {
  return new Promise((resolve) => setTimeout(resolve, ms));
}

function corsHeaders() {
  return {
    "access-control-allow-origin": "*",
    "access-control-allow-methods": "GET,POST,OPTIONS",
    "access-control-allow-headers": "content-type,x-mzj-gateway-secret,x-mersal-webhook-secret",
    "cache-control": "no-store",
  };
}

function json(data, status = 200) {
  return new Response(JSON.stringify(data, null, 2), {
    status,
    headers: { ...corsHeaders(), "content-type": "application/json; charset=utf-8" },
  });
}
